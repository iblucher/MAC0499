\AtBeginBibliography{
  % A margem padrão faz os triângulos indicadores dos itens na
  % bibliografia caírem fora do quadro; aqui, colocamos um espaço
  % adicional para resolver isso. Este comando está também em
  % extras/beamerthemeimeusp-poster.sty, mas sem efeito, não
  % entendi por quê (em apresentacao.tex isso funciona como
  % esperado).
  \addtobeamertemplate{bibliography item}{\hspace*{1.3em}}{}
  % Num poster, a bibliografia pode ficar em tamanho menor
  \footnotesize
}
